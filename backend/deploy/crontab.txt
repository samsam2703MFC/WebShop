# Buddy ⇄ WooCommerce sync + stock reservation cleanup.
# Install:  crontab backend/deploy/crontab.txt   (adapte le chemin ABSOLU ci-dessous)
#
# Remplace /var/www/WebShop/backend par ton chemin réel.

# Pousser prix/stock vers WooCommerce — toutes les 5 min
*/5 * * * * cd /var/www/WebShop/backend && /usr/bin/npm run sync:push >> /var/log/buddy-sync.log 2>&1

# Récupérer les commandes validées depuis WooCommerce — toutes les 5 min
*/5 * * * * cd /var/www/WebShop/backend && /usr/bin/npm run sync:pull >> /var/log/buddy-sync.log 2>&1

# Libérer les réservations de stock expirées — toutes les minutes
* * * * * mysql -u"$WS_DB_USER" -p"$WS_DB_PASS" "$WS_DB_NAME" -e "UPDATE ws_stock_reservations SET released = TRUE WHERE expires_at < NOW() AND released = FALSE;" >> /var/log/buddy-sync.log 2>&1
